Django ORM Cheatsheet

Quick reference for Module 3 & Bonus Module 7 — Models & Advanced ORM. Print-friendly.

Defining a Model

from django.db import models

class Movie(models.Model):
    title        = models.CharField(max_length=200)
    release_year = models.IntegerField()
    rating       = models.FloatField(default=0.0)
    created_at   = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.title
# Table: movies_movie  | auto pk: id

Relationships

# One-to-Many (FK on 'many' side)
director = models.ForeignKey(
    Director,
    on_delete=models.CASCADE,
    related_name='movies')

# Many-to-Many
actors = models.ManyToManyField(
    Actor, related_name='movies')

# One-to-One
movie = models.OneToOneField(
    Movie, on_delete=models.CASCADE)

on_delete options

CASCADE     # delete children too
PROTECT     # block deletion
SET_NULL    # set FK null (needs null=True)
SET_DEFAULT
DO_NOTHING

Traversing relations

movie.director           # forward FK
director.movies.all()    # reverse (related_name)
movie.actors.all()       # M2M forward
movie.actors.add(actor)  # link M2M

Q Objects & F Expressions

from django.db.models import Q, F

filter(Q(a=1) | Q(b=2))    # OR
filter(Q(a=1) & ~Q(b=2))   # AND NOT

# Operate on column values in-DB
Movie.objects.update(
    views=F('views') + 1)

Aggregation & Annotation

from django.db.models import (
    Count, Avg, Sum, Max, Min)

Movie.objects.aggregate(Avg('rating'))
# {'rating__avg': 7.9}

Director.objects.annotate(
    num=Count('movies')
).filter(num__gte=3)

Query Optimization (N+1)

# FK / OneToOne → SQL JOIN
Movie.objects.select_related('director')

# M2M / reverse FK → 2nd query
Movie.objects.prefetch_related('actors')

Use when looping over related objects to avoid one query per row.

QuerySets are lazy

qs = Movie.objects.filter(...)  # no DB hit
list(qs)        # executes now
qs[0]           # executes now
qs.exists()     # executes now
# Chainable before evaluation:
qs.filter(...).order_by(...)[:10]

Advanced Queries — Lesson 03 §5–6

Multiple filter() — the JOIN trap

# Single filter → both conditions on SAME related row
Movie.objects.filter(
    actors__name='Tom', actors__age__gt=30)

# Chained filters → independent JOINs
# (different actors can satisfy each)
Movie.objects.filter(actors__name='Tom')\
             .filter(actors__age__gt=30)

# Safe to chain on the model's own fields:
Movie.objects.filter(release_year=2010)\
             .filter(rating__gte=8.0)

Multi-field order_by + NULLs

# Primary sort, then tie-breaker
Movie.objects.order_by('-rating', 'title')

# NULL handling (portable)
from django.db.models import F
Movie.objects.order_by(F('budget').asc(nulls_last=True))
Movie.objects.order_by(F('rating').desc(nulls_first=True))

# Clear inherited Meta ordering
Movie.objects.filter(...).order_by()

F() — column references & arithmetic

from django.db.models import F

# Compare two columns
Movie.objects.filter(views__gt=F('budget'))

# Atomic increment (no race condition)
Movie.objects.filter(id=1).update(
    views=F('views') + 1)

# Computed annotation
from django.db.models import ExpressionWrapper, BigIntegerField
Movie.objects.annotate(
    profit=ExpressionWrapper(
        F('revenue') - F('budget'),
        output_field=BigIntegerField()))

values() / values_list() + GROUP BY

# List of dicts (projection)
Movie.objects.values('title', 'rating')

# Flat list of a single field
ids = Movie.objects.values_list('id', flat=True)

# GROUP BY: values() before annotate()
from django.db.models import Count, Avg
Movie.objects.values('release_year')\
             .annotate(count=Count('id'))\
             .order_by('release_year')

Movie.objects.values('director__name')\
             .annotate(avg=Avg('rating'))

annotate() + filter (HAVING)

from django.db.models import Count, Avg, Q

# Per-row count; filter on it → HAVING clause
Director.objects.annotate(
    n=Count('movies')
).filter(n__gte=3).order_by('-n')

# Conditional count (only good movies)
Director.objects.annotate(
    good=Count('movies',
        filter=Q(movies__rating__gte=8)))

only() / defer()

# Load ONLY these columns (others trigger extra query)
Movie.objects.only('title', 'rating')

# Load everything EXCEPT these columns
Movie.objects.defer('synopsis')

# Safe pattern: only() fields you actually use
for m in Movie.objects.only('title'):
    print(m.title)   # no extra query
    # m.release_year → extra query!

distinct()

# M2M filter can return duplicate parent rows
# → distinct() adds SELECT DISTINCT
Movie.objects.filter(
    actors__name__startswith='Tom'
).distinct()

# Also useful after prefetch + annotation
Director.objects.filter(
    movies__rating__gt=8
).distinct()

get_or_create / update_or_create

# Returns (object, was_created: bool)
movie, created = Movie.objects.get_or_create(
    title='Inception',           # lookup (WHERE)
    defaults={'release_year': 2010, 'rating': 8.8})

# Upsert — defaults applied on create AND update
movie, created = Movie.objects.update_or_create(
    title='Inception',
    defaults={'rating': 9.0})

bulk_create / bulk_update

# One INSERT for all rows (skips signals!)
Movie.objects.bulk_create([
    Movie(title='Dune', release_year=2021),
    Movie(title='Dune 2', release_year=2024),
], batch_size=500, ignore_conflicts=True)

# One UPDATE for all rows (name changed fields)
for m in movies: m.rating += 0.1
Movie.objects.bulk_update(movies, ['rating'])

Subquery / OuterRef / Exists

from django.db.models import OuterRef, Subquery, Exists

# Annotate each Director with their latest movie title
latest = Movie.objects.filter(
    director=OuterRef('pk')
).order_by('-release_year').values('title')[:1]

Director.objects.annotate(latest_title=Subquery(latest))

# Efficient membership filter
has_hit = Movie.objects.filter(
    director=OuterRef('pk'), rating__gt=9)
Director.objects.filter(Exists(has_hit))
