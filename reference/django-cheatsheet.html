DRF — Serializer

from rest_framework import serializers

class BookSerializer(serializers.ModelSerializer):
    class Meta:
        model  = Book
        fields = ['id', 'title', 'author']
        # or: fields = '__all__'
        # or: exclude = ['internal_field']

# Serialize (→ JSON)
s = BookSerializer(book_instance)
s.data  # dict

# Serialize many
s = BookSerializer(queryset, many=True)

# Deserialize (← JSON) + validate
s = BookSerializer(data=request.data)
s.is_valid(raise_exception=True)
s.save()  # calls create() or update()

DRF — Function-based View

from rest_framework.decorators import api_view
from rest_framework.response import Response
from rest_framework import status

@api_view(['GET', 'POST'])
def book_list(request):
    if request.method == 'GET':
        books = Book.objects.all()
        s = BookSerializer(books, many=True)
        return Response(s.data)

    if request.method == 'POST':
        s = BookSerializer(data=request.data)
        s.is_valid(raise_exception=True)
        s.save()
        return Response(s.data, status=status.HTTP_201_CREATED)

urls.py — URL Routing

# myapp/urls.py
from django.urls import path
from . import views

urlpatterns = [
    path('api/books/',     views.book_list),
    path('api/books/<int:pk>/', views.book_detail),
]

# project/urls.py  (root)
from django.urls import path, include

urlpatterns = [
    path('', include('myapp.urls')),
]

Common HTTP Status Codes

200 OK            — successful GET / PUT
201 Created       — successful POST
204 No Content    — successful DELETE
400 Bad Request   — validation failed
401 Unauthorized  — not authenticated
403 Forbidden     — not authorised
404 Not Found     — resource missing
500 Server Error  — unhandled exception
