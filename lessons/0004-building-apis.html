1. REST API Basics

A REST API exposes resources (like movies) over HTTP. The HTTP method says what to do; the URL says to which resource.

MethodIntentExampleHas body?
GETReadGET /movies/1/No
POSTCreatePOST /movies/Yes
PUTReplace / updatePUT /movies/1/Yes
DELETERemoveDELETE /movies/1/No

Request & Response

A request carries a method, URL, headers, and (sometimes) a JSON body. A response carries a status code, headers, and a JSON body.

2. The Plain Django Pattern (HttpResponse)

Parts of the production codebase return JSON using plain Django — no DRF. You build the response yourself with JsonResponse (a thin wrapper over HttpResponse that serialises a dict to JSON). 

# movies/views.py  — plain Django
import json
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
from django.shortcuts import get_object_or_404
from .models import Movie

def movie_detail(request, movie_id):
    movie = get_object_or_404(Movie, id=movie_id)
    data = {
        'id': movie.id,
        'title': movie.title,
        'release_year': movie.release_year,
        'rating': movie.rating,
    }
    return JsonResponse(data)                      # 200 OK

@csrf_exempt
def movie_create(request):
    if request.method != 'POST':
        return JsonResponse({'error': 'method not allowed'}, status=405)
    payload = json.loads(request.body)             # parse JSON body manually
    movie = Movie.objects.create(
        title=payload['title'],
        release_year=payload['release_year'],
    )
    return JsonResponse({'id': movie.id}, status=201)
Notice the manual work
You hand-build the dict, parse request.body yourself, and there's no validation. This is exactly the pain that DRF removes.

3. The DRF Pattern (Serializer + APIView + Response)

Django REST Framework adds three pieces that eliminate the boilerplate above. The rest of the production codebase uses this style.

Serializer — Model ↔ JSON + validation

# movies/serializers.py
from rest_framework import serializers
from .models import Movie

class MovieSerializer(serializers.ModelSerializer):
    class Meta:
        model  = Movie
        fields = ['id', 'title', 'release_year', 'rating']

APIView — class-based view, one method per HTTP verb

# movies/views.py  — DRF
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from django.shortcuts import get_object_or_404
from .models import Movie
from .serializers import MovieSerializer

class MovieDetail(APIView):
    def get(self, request, movie_id):
        movie = get_object_or_404(Movie, id=movie_id)
        serializer = MovieSerializer(movie)
        return Response(serializer.data)            # 200 OK

    def put(self, request, movie_id):
        movie = get_object_or_404(Movie, id=movie_id)
        serializer = MovieSerializer(movie, data=request.data)
        serializer.is_valid(raise_exception=True)   # 400 if invalid
        serializer.save()
        return Response(serializer.data)

    def delete(self, request, movie_id):
        movie = get_object_or_404(Movie, id=movie_id)
        movie.delete()
        return Response(status=status.HTTP_204_NO_CONTENT)

class MovieList(APIView):
    def post(self, request):
        serializer = MovieSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)   # validation built-in!
        serializer.save()
        return Response(serializer.data, status=status.HTTP_201_CREATED)

Wire up the URLs

# movies/urls.py
from django.urls import path
from .views import MovieDetail, MovieList

urlpatterns = [
    path('movies/', MovieList.as_view()),
    path('movies/<int:movie_id>/', MovieDetail.as_view()),
]
