from django_swagger_utils.utils.test import CustomAPITestCase

from fb_post.models import Comment, Post
from . import APP_NAME, OPERATION_NAME, REQUEST_METHOD, URL_SUFFIX
import json
REQUEST_BODY = """
{
    "comment_content": "Create New Comment"
}
"""

TEST_CASE = {
    "request": {
        "path_params": {"post_id": "1"},
        "query_params": {},
        "header_params": {},
        "securities": {"oauth": {"tokenUrl": "http://auth.ibtspl.com/oauth2/", "flow": "password", "scopes": ["superuser"], "type": "oauth2"}},
        "body": REQUEST_BODY,
    },
}


class TestCase01AddCommentToAPostAPITestCase(CustomAPITestCase):
    app_name = APP_NAME
    operation_name = OPERATION_NAME
    request_method = REQUEST_METHOD
    url_suffix = URL_SUFFIX
    test_case_dict = TEST_CASE

    def setupUser(self, username, password):
        super(TestCase01AddCommentToAPostAPITestCase, self).setupUser(
            username=username,
            password=password
        )
        Post.objects.create(user_id=self.foo_user.id, post_content='test_post')

    def test_case(self):
        api_response = self.default_test_case()  # Returns response object.
        # Which can be used for further response object checks.
        # Add database state checks here.

        response_data = json.loads(api_response.content)
        comment_id = response_data['comment_id']
        comment = Comment.objects.get(id=comment_id)

        # add to snapshot
        self.assert_match_snapshot(
            name="comment_content",
            value=comment.comment_content
        )

        self.assert_match_snapshot(
            name="post_id",
            value=comment.post_id
        )
