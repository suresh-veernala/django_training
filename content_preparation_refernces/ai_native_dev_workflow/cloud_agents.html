You should see 3 urls now with 2 labeled as (push).

Setup Script 
#!/usr/bin/env bash
set -euo pipefail

export PATH="$HOME/.local/bin:$PATH"

# Must run from repo root
if [ ! -f pyproject.toml ]; then
  echo "error: run this script from the repository root (pyproject.toml not found here)" >&2
  exit 1
fi

# --- Debian/Ubuntu: build deps for mysqlclient, Kerberos (requests-kerberos / gssapi), etc. ---
install_native_deps() {
  if ! command -v apt-get >/dev/null 2>&1; then
    return 0
  fi
  local runner=""
  if [ "$(id -u)" -ne 0 ]; then
    runner="sudo"
  fi
  $runner apt-get update -qq
  DEBIAN_FRONTEND=noninteractive $runner apt-get install -y -qq \
    build-essential \
    pkg-config \
    python3-dev \
    default-libmysqlclient-dev \
    libkrb5-dev
}
install_native_deps

# Ensure AWS CLI is available
if ! command -v aws &> /dev/null; then
  curl -s "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "/tmp/awscliv2.zip"
  unzip -o /tmp/awscliv2.zip -d /tmp/ > /dev/null
  sudo /tmp/aws/install --update > /dev/null
  rm -f /tmp/awscliv2.zip
fi

# --- AWS profile (nw-tech) from environment ---
mkdir -p ~/.aws
python3 -c "
import os
key_id = os.environ.get('AWS_ACCESS_KEY_ID', '')
secret = os.environ.get('AWS_SECRET_ACCESS_KEY', '')
with open(os.path.expanduser('~/.aws/credentials'), 'w') as f:
    f.write(f'[nw-tech]\naws_access_key_id = {key_id}\naws_secret_access_key = {secret}\n')
"
cat > ~/.aws/config << 'AWSCONFIG'
[profile nw-tech]
region = ap-south-1
output = json
AWSCONFIG

# --- CodeArtifact token via assume-role ---
ASSUMED=$(aws sts assume-role \
  --role-arn "arn:aws:iam::110916805156:role/CursorAgentCodeArtifactReadPermissions" \
  --role-session-name "cursor-session" \
  --profile nw-tech \
  --output json)
AWS_AK=$(echo "$ASSUMED" | python3 -c "import sys,json; print(json.load(sys.stdin)['Credentials']['AccessKeyId'])")
AWS_SK=$(echo "$ASSUMED" | python3 -c "import sys,json; print(json.load(sys.stdin)['Credentials']['SecretAccessKey'])")
AWS_ST=$(echo "$ASSUMED" | python3 -c "import sys,json; print(json.load(sys.stdin)['Credentials']['SessionToken'])")
CODEARTIFACT_AUTH_TOKEN=$(AWS_ACCESS_KEY_ID="$AWS_AK" AWS_SECRET_ACCESS_KEY="$AWS_SK" AWS_SESSION_TOKEN="$AWS_ST" \
  aws codeartifact get-authorization-token \
    --domain nxtwave \
    --region ap-south-1 \
    --query authorizationToken \
    --output text)

pip install poetry==2.2.1

mkdir -p ~/.config/pypoetry
cat > ~/.config/pypoetry/auth.toml << EOF
[http-basic.codeartifact]
username = "aws"
password = "$CODEARTIFACT_AUTH_TOKEN"
EOF

# --- Poetry: always use in-project .venv (never system Python) ---
poetry config virtualenvs.create true
poetry config virtualenvs.in-project true
poetry env use python3
poetry install --with dev,lint,test --no-root

# Check and install boto3 in venv
.venv/bin/python -c "import boto3" 2>/dev/null || .venv/bin/pip install boto3

# Setup agent-toolkit-backend AWS profile
python3 -c "
import os
key_id = os.environ.get('BACKEND_TOOLKIT_AWS_ACCESS_KEY', '')
secret = os.environ.get('BACKEND_TOOLKIT_AWS_SECRET_KEY', '')
with open(os.path.expanduser('~/.aws/credentials'), 'a') as f:
    f.write(f'[agent-toolkit-backend]\naws_access_key_id = {key_id}\naws_secret_access_key = {secret}\n')
"

cat >> ~/.aws/config << 'AWSCONFIG'
[profile agent-toolkit-backend]
region = ap-south-1
output = json
AWSCONFIG

# Download sync script and run
TMP_DIR=$(mktemp -d)
aws s3 cp s3://nw-agent-toolkit-backend/server/nw-agent-toolkit-server-sync.py "$TMP_DIR/nw-agent-toolkit-server-sync.py" --profile agent-toolkit-backend
.venv/bin/python "$TMP_DIR/nw-agent-toolkit-server-sync.py" --profile agent-toolkit-backend

source .venv/bin/activate

if command -v pre-commit >/dev/null 2>&1; then
  pre-commit install
  pre-commit install --hook-type pre-push
fi

echo "Done"

# Clickup

  • Clickup is already connected to prasanth's cursor account for whole workspace.
  • Cursor is already integrated into Clickup. You can validate it by opening any of your task and type @cursor and see if the agent is in suggestions.
  • How to use it:
    • Open any task with clear descriptions like issue or clear backend description on what to do
    • Type @cursor in the comment tab and copy below prompt templates.
    • Add if you have any additional instructions related to the task
    • You can view the task by clicking on the link the agent returned.

Invocation Format:

@Cursor

repo=nxtwave-tech/ai_mock_interview_poc_backend

{prompt}

The description is auto attached. You need to send additional instructions only.

For Issues, use:

\n
Resolve the above issue by following the guidelines in 
```
.cursor/commands/resolve-issue.md
```
- Avoid speculations and overthinking of unrealistic scenarios.
- Avoid adding preventive safeguards or hacky fixes just for the sake of it.
- Avoid tests modification unless the current issue you have fixed is a valid branch that needs explicit coverage.
- If the root cause is clear, create PR. Otherwise, mention the analysis and the next steps.

For straightforward tasks where no additional ambiguity exists for agent, use:

\n
Plan and implement the code for the above requirement using the approach:
```
approach in your own words
```
by following the guidelines in
```
.cursor/commands/implement-with-plan.md
```
- Follow codebase patterns strictly when ambiguous follow `skill-map`.
- Write Optimized and efficient code. Avoid Re-fetching data or writing orm queries or external service calls in loops.
- Follow existing codebase pattern namings and strip requirement specific namings.
- Avoid default arguments for dtos, method args etc.
- Write readable and minimal code and avoid complex code patterns.
- If you are unable to identify the right direction or blocked related to ambiguity, stop the execution and ask the questions. Otherwise, create PR.

  • Fill the below prompt. Choose the model you prefer

  • Go to https://nxtwave.sentry.io/settings/developer-settings/ and create a internal integration with convention -> Cursor Agent.
  • In webhook url field, paste the lambda url given to you. Fill the url query param and hash the auth token copied using below function
import hmac
import hashlib
import base64

def hash_with_secret(token, secret):
    key = secret.encode('utf-8')
    message = token.encode('utf-8')
    
    signature = hmac.new(key, message, hashlib.sha256).digest()
    return base64.urlsafe_b64encode(signature).decode('utf-8')
  • Toggle Alert Rule Action to Enabled state.
  • In permissions, give Read access to Issue & Event
  • Save the integration
  • In Alerts section for your project, add this new integration and test with a sample event or a issue.
  • Check whether cloud agent is invoked or not in Run History of your automation
  • Prefer the direct automation with assigned trigger over the webhook automation, as this flow again involves custom lambda invocation and triggers based on alert only which can sometimes trigger for unrelated issues.
  • Ensure multiple agents are not triggered in this flow with right alert rules.
  • Note: The payload to lambda webhook must have the key stacktrace otherwise agent won't be invoked.
  • If you face any issues in triggering cloud agents in this flow, can debug lambda - sentry-cursor-webhook-triggeer in aws alpha.
