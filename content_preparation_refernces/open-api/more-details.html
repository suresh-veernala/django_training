        # Open API Specification

        # Defining query parameters

        DETAILS 
        {
	"paths": {
		"/persons/": {
			"get": {
			    ...
				"parameters": [
					{
						"name": "offset",
						"in": "query",
						"description": "Some description for offset",
						"type": "integer"
					},
					{
						"name": "limit",
						"in": "query",
						"description": "Some description for limit",
						"type": "integer"
					}
				],
				...
			}
		}
	}
}

        The above specification can help us specify urs like /persons?offset=2&limit=20.

        # Defining a path parameter

        DETAILS 
        {
	"paths": {
		"/persons/{username}/": {
			"get": {
			    ...
				"parameters": [
					{
						"name": "username",
						"in": "path",
						"required": true,
						"description": "The person's username",
						"type": "string"
					}
				],
				...
			}
		}
	}
}

        The above specification can help us specify urs like /persons/ryan/.

        # Frequently Used Types:

        Most commonly used as value for type are

        • object
        • string
        • enum
        • int
        • number
        • array

        We can optionally specify formats to some types which are mentioned at respective sections if any available. Formats are used while generating client modules (mainly for typed languages)

        Check reference for an exhaustive list of types and their formats.

        # Specifying String values in Schema

        We need to define type as string. Commonly used values for format are date and datetime.

        DETAILS 
        {
    "definitions": {
        "User": {
            "type": "object",
            "properties": {
                "username": {
                    "type": "string"
                },
                "date_of_birth": {
                    "type": "string",
                    "format": "date"
                }
            },
            "required": [
                "username",
                "date_of_birth"
            ]
        }
    }
}

        # Specifying Enum values in Schema

        DETAILS 
        {
    "definitions": {
        "User": {
            "type": "object",
            "properties": {
                "username": {
                    "type": "string"
                },
                "gender": {
                    "type": "string",
                    "enum": [
                      "MALE",
                      "FEMALE"
                    ]
                }
            },
            "required": [
                "username",
                "gender"
            ]
        }
    }
}

        # Specifying Integer values in Schema

        We need to define type as integer. Commonly values for format are int32 and int64.

        DETAILS 
        {
    "definitions": {
        "User": {
            "type": "object",
            "properties": {
                "username": {
                    "type": "string"
                },
                "age": {
                    "type": "integer",
                    "format": "int64"
                }
            },
            "required": [
                "username",
                "age"
            ]
        }
    }
}

        # Specifying floating values in Schema

        We need to define type as number.Commonly used values for format are float and double.

        DETAILS 
        {
    "definitions": {
        "User": {
            "type": "object",
            "properties": {
                "username": {
                    "type": "string"
                },
                "score": {
                    "type": "number",
                    "format": "float"
                }
            },
            "required": [
                "score",
                "username"
            ]
        }
    }
}

        # Specifying Arrays in Schema

        We need to define type as array

        DETAILS 
        {
    "definitions": {
        "User": {
            "type": "object",
            "properties": {
                "username": {
                    "type": "string"
                },
                "age": {
                    "type": "integer",
                    "format": "int64"
                }
            },
            "required": [
                "username",
                "age"
            ]
        },
        "Group": {
            "type": "object",
            "properties": {
                "members": {
                    "type": "array",
                    "items": {
                        "$ref": "#/definitions/User"
                    }
                }
            },
            "required": [
                "members"
            ]
        }
    }
}

        # Defining responses

        Don’t forget to define responses for this new operation.

        {
	"responses": {
		"204": {
			"description": "Persons successfully created."
		},
		"400": {
			"description": "Persons couldn't have been created."
		}
	}
}

        # API description

        Then we give some informations about our API with info: the API’s version (not to be confused with the specification version and the file version) a title and an optional description.

            {
	"info": {
		"version": "1.0.0",
		"title": "Simple API",
		"description": "A simple API to learn how to write OpenAPI Specification"
	}
}
